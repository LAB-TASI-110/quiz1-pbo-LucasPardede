package model;

import java.util.ArrayList;


public class solution {

    private ArrayList<String> listMenu = new ArrayList<>();
    private ArrayList<Integer> listHarga = new ArrayList<>();
    private ArrayList<Integer> listPorsi = new ArrayList<>();
    private int total = 0;

    public void tambahPesanan(String kode, int porsi) {
        String namaMenu = "";
        int harga = 0;

        switch (kode) {
            case "NGS":
                namaMenu = "Nasi Goreng Spesial";
                harga = 25000;
                break;
            case "AP":
                namaMenu = "Ayam Penyet";
                harga = 20000;
                break;
            case "GG":
                namaMenu = "Gado-Gado";
                harga = 15000;
                break;
            default:
                System.out.println("Kode tidak valid");
                return;
        }

        listMenu.add(namaMenu);
        listHarga.add(harga);
        listPorsi.add(porsi);

        total += harga * porsi;
    }

    public void cetakStruk() {
        System.out.println("===== STRUK PEMBELIAN =====");

        for (int i = 0; i < listMenu.size(); i++) {
            System.out.println(listMenu.get(i) + " x" + listPorsi.get(i) +
                    " = " + (listHarga.get(i) * listPorsi.get(i)));
        }

        System.out.println("----------------------------");
        System.out.println("Total : " + total);

        if (total >= 200000) {
            System.out.println("Selamat! Anda mendapatkan Kupon Kuning");
        } else if (total >= 100000) {
            System.out.println("Selamat! Anda mendapatkan Kupon Biru");
        } else {
            System.out.println("Belum mendapatkan kupon");
        }
    }
}
